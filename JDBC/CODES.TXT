    void loadTable() {
        try {
            model.setRowCount(0);

            ResultSet rs = con.prepareStatement("SELECT * FROM warehouse_copy").executeQuery();

            while (rs.next()) {
                model.addRow(new Object[]{
                        rs.getInt(1),
                        rs.getString(2),
                        rs.getString(3),
                        rs.getInt(4),
                        rs.getString(5)
                });
            }

        } catch (SQLException e) {
            JOptionPane.showMessageDialog(this, "Load Error: " + sqlFriendlyMessage(e));
        } catch (Exception e) {
            JOptionPane.showMessageDialog(this, "Load Error: " + e.getMessage());
        }
    }

        txtId.setText("");
        txtName.setText("");
        txtCity.setText("");
        txtCapacity.setText("");
        txtStatus.setText("");
    }

    // Translate common SQLExceptions to user-friendly messages
    String sqlFriendlyMessage(SQLException e) {
        if (e == null) return "Unknown database error.";
        String msg = e.getMessage();
        if (msg == null) msg = "";

        if (msg.contains("ORA-00001") || msg.toLowerCase().contains("unique constraint") ) {
            return "Duplicate value violates a unique constraint (e.g., branch_id or branch_name already exists).";
        }
        if (msg.contains("ORA-01400") || msg.toLowerCase().contains("cannot be null")) {
            return "A required column was null. Please provide all required values (ID, Name, etc.).";
        }
        if (msg.contains("ORA-02292")) {
            return "Cannot delete: child records exist referencing this branch. Remove dependent records first.";
        }
        if (msg.contains("ORA-02291")) {
            return "Foreign key constraint failed: referenced record not found.";
        }

        return "Database error (" + e.getErrorCode() + "): " + msg;
    }

    public static void main(String[] args) {
        new WarehouseBranchForm();
    }
}




import java.sql.*;

public class WarehouseBranch {

    public static void main(String[] args) {

        try {
            con = DBConnection.getConnection();

    static void insert() {
        try {
            System.out.print("Enter ID: ");
            int id = sc.nextInt();
            sc.nextLine();

            System.out.print("Enter Name: ");
            String name = sc.nextLine();

            System.out.print("Enter City: ");
            String city = sc.nextLine();

            System.out.print("Enter Capacity: ");
            int capacity = sc.nextInt();
            sc.nextLine();

            System.out.print("Enter Status: ");
            String status = sc.nextLine();

            PreparedStatement ps = con.prepareStatement(
                    "INSERT INTO warehouse_copy VALUES(?,?,?,?,?)");

            ps.setInt(1, id);
            ps.setString(2, name);
            ps.setString(3, city);
            ps.setInt(4, capacity);
            ps.setString(5, status);

            ps.executeUpdate();

            System.out.println("Inserted Successfully");

        } catch (SQLException e) {
            System.out.println("Insert Error: " + sqlFriendlyMessage(e));
        } catch (Exception e) {
            System.out.println("Insert Error: " + e.getMessage());
        }
    }

    
    static void update() {
        try {
            System.out.print("Enter ID to update: ");
            int id = sc.nextInt();
            sc.nextLine();

            PreparedStatement check = con.prepareStatement(
                    "SELECT * FROM warehouse_copy WHERE branch_id=?");
            check.setInt(1, id);
            ResultSet rs = check.executeQuery();

            if (!rs.next()) {
                System.out.println(" Record Not Found");
                return;
            }

            String currentName = rs.getString("branch_name");
            String currentCity = rs.getString("city");
            int currentCapacity = rs.getInt("capacity");
            String currentStatus = rs.getString("status");

            System.out.print("Enter New Name (leave you dont want to update): ");
            String name = sc.nextLine().trim();
            if (name.isEmpty()) name = currentName;

            System.out.print("Enter New City (leave you dont want to update): ");
            String city = sc.nextLine().trim();
            if (city.isEmpty()) city = currentCity;

            System.out.print("Enter New Capacity (leave you dont want to update): ");
            String capInput = sc.nextLine().trim();
            int capacity = currentCapacity;
            if (!capInput.isEmpty()) {
                capacity = Integer.parseInt(capInput);
            }

            System.out.print("Enter New Status (leave you dont want to update): ");
            String status = sc.nextLine().trim();
            if (status.isEmpty()) status = currentStatus;

            StringBuilder sql = new StringBuilder("UPDATE warehouse_copy SET ");
            List<Object> params = new ArrayList<>();

            if (!name.equals(currentName)) {
                sql.append("branch_name=?, ");
                params.add(name);
            }
            if (!city.equals(currentCity)) {
                sql.append("city=?, ");
                params.add(city);
            }
            if (capacity != currentCapacity) {
                sql.append("capacity=?, ");
                params.add(capacity);
            }
            if (!status.equals(currentStatus)) {
                sql.append("status=?, ");
                params.add(status);
            }

            if (params.isEmpty()) {
                System.out.println("No changes made.");
                return;
            }

            sql.setLength(sql.length() - 2);
            sql.append(" WHERE branch_id=?");
            params.add(id);

            PreparedStatement ps = con.prepareStatement(sql.toString());
            for (int i = 0; i < params.size(); i++) {
                ps.setObject(i + 1, params.get(i));
            }

            ps.executeUpdate();

            System.out.println("Updated Successfully");

        } catch (SQLException e) {
            System.out.println("Update Error: " + sqlFriendlyMessage(e));
        } catch (Exception e) {
            System.out.println("Update Error: " + e.getMessage());
        }
    }

    static void delete() {
        try {
            System.out.print("Enter ID to delete: ");
            int id = sc.nextInt();

            PreparedStatement check = con.prepareStatement(
                    "SELECT * FROM warehouse_copy WHERE branch_id=?");
            check.setInt(1, id);
            ResultSet rs = check.executeQuery();

            if (!rs.next()) {
                System.out.println(" Record Not Found");
                return;
            }

            PreparedStatement ps = con.prepareStatement(
                    "DELETE FROM warehouse_copy WHERE branch_id=?");

            ps.setInt(1, id);

        } catch (SQLException e) {
            System.out.println("Delete Error: " + sqlFriendlyMessage(e));
        } catch (Exception e) {

    // Normalize an email for duplicate comparison.
    // - Lower-cases domain
    // - Treats googlemail.com as gmail.com
    // - For gmail.com: removes dots from local-part and strips anything after '+'
    // - For other domains: compares case-insensitively (lower-cased)
    String normalizeEmailForComparison(String email) {
        if (email == null) return "";
        String e = email.trim();
        int at = e.lastIndexOf('@');
        if (at < 0) return e.toLowerCase();
        String local = e.substring(0, at);
        String domain = e.substring(at + 1).toLowerCase();

        if (domain.equals("googlemail.com")) domain = "gmail.com";

        if (domain.equals("gmail.com")) {
            // remove anything after +
            int plus = local.indexOf('+');
            if (plus >= 0) local = local.substring(0, plus);
            // remove dots
            local = local.replace(".", "");
            return local.toLowerCase() + "@" + domain;
        }

        return local.toLowerCase() + "@" + domain;
    }

    // General email validation that supports multiple domains and delegates
    // additional checks for Gmail local-part requirements.
    boolean isValidEmail(String email) {
        if (email == null) return false;
        String e = email.trim();
        // basic email pattern
        String basic = "^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\\.[A-Za-z]{2,}$";
        if (!e.matches(basic)) return false;
        int at = e.lastIndexOf('@');
        if (at < 0) return false;
        String domain = e.substring(at + 1).toLowerCase();

        // treat googlemail as gmail
        if (domain.equals("googlemail.com")) domain = "gmail.com";

        // Gmail-specific checks: local-part must contain at least one letter
        if (domain.equals("gmail.com")) {
            String local = e.substring(0, at);
            int plus = local.indexOf('+');
            if (plus >= 0) local = local.substring(0, plus);
            local = local.replace(".", "");
            return local.matches(".*[A-Za-z].*");
        }

        // other domains: basic format already validated
        return true;
    }

   
    boolean isValidGmail(String email) {
        if (email == null) return false;
     
        String gmailRegex = "^(?=.{1,64}@gmail\\.com)(?=[^@]*[A-Za-z])[A-Za-z0-9._%+-]+@gmail\\.com$";
        return email.matches(gmailRegex);
    }

    boolean isValidPhone(String phone) {
        if (phone == null) return false;
        String digits = phone.replaceAll("\\D", "");
        return digits.matches("\\d{10}");
    }

    boolean isValidPassword(String password) {
        if (password == null) return false;
        String pwdRegex = "(?=.*\\d)(?=.*[^A-Za-z0-9]).{8,}";
        return password.matches(pwdRegex);
    }

        }
    }

import javax.swing.*;
import javax.swing.table.DefaultTableModel;
import java.awt.*;
import java.sql.*;

public class ProductDisplayPage extends JFrame {

    JTable table;
    DefaultTableModel model;
    Connection con;

    public ProductDisplayPage() {

        setTitle("Product Display");
        setSize(900, 450);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLayout(new BorderLayout());

      
        JLabel titleLabel = new JLabel("Product Information", SwingConstants.CENTER);
        titleLabel.setFont(new Font("Arial", Font.BOLD, 20));
        add(titleLabel, BorderLayout.NORTH);

   
        model = new DefaultTableModel(
                new String[]{"Product ID", "Product Name", "Category", "Price", "Stock Quantity", "Description"}, 0);

        table = new JTable(model);
        add(new JScrollPane(table), BorderLayout.CENTER);

       
        JPanel panel = new JPanel();

        JButton btnSearchName = new JButton("Search by Name");
        JButton btnSearchCategory = new JButton("Search by Category");
        JButton btnSearchId = new JButton("Search by ID");

        panel.add(btnSearchName);
        panel.add(btnSearchCategory);
        panel.add(btnSearchId);

        add(panel, BorderLayout.SOUTH);

        con = DBConnection.getConnection();

        if (con == null) {
            JOptionPane.showMessageDialog(this, "❌ Database Connection Failed");
        } else {
            System.out.println("✅ Connected");
            loadProducts();
        }

        btnSearchName.addActionListener(e -> {
            try {
                String name = JOptionPane.showInputDialog(this, "Enter product name:");
                if (name == null || name.trim().isEmpty()) return;

                name = name.trim();

                model.setRowCount(0);

                PreparedStatement ps = con.prepareStatement(
                        "SELECT * FROM products1 WHERE LOWER(product_name) LIKE ?");

                ps.setString(1, "%" + name.toLowerCase() + "%");

                ResultSet rs = ps.executeQuery();

                boolean found = false;
                while (rs.next()) {
                    addRow(rs);
                    found = true;
                }

                if (!found) {
                    JOptionPane.showMessageDialog(this, "No product found");
                    loadProducts();
                }

            } catch (Exception ex) {
                ex.printStackTrace();
            }
        });

       
        btnSearchCategory.addActionListener(e -> {
            try {
                String cat = JOptionPane.showInputDialog(this, "Enter category:");
                if (cat == null || cat.trim().isEmpty()) return;

                cat = cat.trim();

                model.setRowCount(0);

                PreparedStatement ps = con.prepareStatement(
                        "SELECT * FROM products1 WHERE LOWER(category) LIKE ?");

                ps.setString(1, "%" + cat.toLowerCase() + "%");

                ResultSet rs = ps.executeQuery();

                boolean found = false;
                while (rs.next()) {
                    addRow(rs);
                    found = true;
                }

                if (!found) {
                    JOptionPane.showMessageDialog(this, "No product found");
                    loadProducts();
                }

            } catch (Exception ex) {
                ex.printStackTrace();
            }
        });

        btnSearchId.addActionListener(e -> {
            String idStr = JOptionPane.showInputDialog(this, "Enter Product ID:");
            if (idStr == null || idStr.trim().isEmpty()) return;

            idStr = idStr.trim();

            int id;
            try {
                id = Integer.parseInt(idStr);
            } catch (NumberFormatException nfe) {
                JOptionPane.showMessageDialog(this, "Invalid Product ID");
                return;
            }

            try {
                model.setRowCount(0);

                PreparedStatement ps = con.prepareStatement(
                        "SELECT * FROM products1 WHERE product_id = ?");

                ps.setInt(1, id);

                ResultSet rs = ps.executeQuery();

                if (rs.next()) {
                    addRow(rs);
                } else {
                    JOptionPane.showMessageDialog(this, "Product  not found");
                    loadProducts();
                }

            } catch (Exception ex) {
                ex.printStackTrace();
            }
        });

        setVisible(true);
    }

    void loadProducts() {
        try {
            model.setRowCount(0);

            PreparedStatement ps = con.prepareStatement(
                    "SELECT * FROM products1 ORDER BY product_id");

            ResultSet rs = ps.executeQuery();

            while (rs.next()) {
                addRow(rs);
            }

        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    void addRow(ResultSet rs) throws Exception {

        String desc = "";

        Object obj = rs.getObject("description");

        if (obj != null) {
            if (obj instanceof Clob) {
                Clob clob = (Clob) obj;
                desc = clob.getSubString(1, (int) clob.length());
            } else {
                desc = obj.toString();
            }
        }

        model.addRow(new Object[]{
                rs.getInt("product_id"),
                rs.getString("product_name"),
                rs.getString("category"),
                rs.getDouble("price"),
                rs.getInt("stock_quantity"),
                desc
        });
    }

    public static void main(String[] args) {
        SwingUtilities.invokeLater(() -> new ProductDisplayPage());
    }
}
