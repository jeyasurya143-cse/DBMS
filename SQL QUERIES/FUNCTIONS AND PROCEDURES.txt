SQL> CREATE OR REPLACE PROCEDURE highest_price_per_category IS
  2      CURSOR c_cats IS SELECT DISTINCT category_id FROM product;
  3      v_cat_id NUMBER;
  4  BEGIN
  5      OPEN c_cats;
  6      FETCH c_cats INTO v_cat_id;
  7      WHILE c_cats%FOUND LOOP
  8          -- Nested loop to find products matching the max price for this category
  9          FOR rec IN (SELECT product_name, price FROM product
 10                      WHERE category_id = v_cat_id
 11                      AND price = (SELECT MAX(price) FROM product WHERE category_id = v_cat_id))
 12          LOOP
 13              DBMS_OUTPUT.PUT_LINE('Cat '||v_cat_id||': '||rec.product_name||' ('||rec.price||')');
 14          END LOOP;
 15
 16          FETCH c_cats INTO v_cat_id;
 17      END LOOP;
 18      CLOSE c_cats;
 19  END;
